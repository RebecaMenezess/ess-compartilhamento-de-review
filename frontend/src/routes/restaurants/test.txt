const getUser = () => {
    
        useEffect(() => {
            const userData = getUserInfoFromToken();
            if (userData && userData.userId) {
                // Assuming you have an endpoint to fetch user profile by ID
                axios.get(`${API_BASE}/users/${userData.userId}`)
                    .then(response => {
                        setUser(response.data);
                    })
                    .catch(error => {
                        console.error("Failed to fetch user data", error);
                    });
            }
        }, []);
    }
